SELECT * FROM LIBRO WHERE titulo LIKE 'El%';   --Like--
SELECT * FROM LIBRO WHERE precio BETWEEN 30 AND 50;--Between--
SELECT * FROM LIBRO WHERE autor IN ('Stephen King', 'Isaac Asimov', 'Dan Brown');--In--
SELECT * FROM LIBRO WHERE autor IS NULL;--IS NULL--
SELECT autor, COUNT(*) AS CantidadLibros, AVG(precio) AS PrecioPromedio--COUNT--
   FROM LIBRO
   GROUP BY autor;
SELECT autor, SUM(stock) AS TotalStock--HAVING--
   FROM LIBRO
   GROUP BY autor
   HAVING SUM(stock) > 10;

SELECT titulo, precio,--CASE--
          CASE 
              WHEN precio < 30 THEN 'Económico'
              WHEN precio BETWEEN 30 AND 55 THEN 'Estándar'
              ELSE 'Premium'
          END AS CategoriaPrecio
   FROM LIBRO;
