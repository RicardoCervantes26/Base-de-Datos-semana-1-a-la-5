GO

-- 2. Decirle a SQL que use esta base de datos
USE Ejemplo;
GO

-- 3. Crear la tabla "Padre" (Categorías)
CREATE TABLE TIPO (
    idtipo CHAR(3) PRIMARY KEY, -- Código corto como 'LIB', 'REV'
    descripcion VARCHAR(50) NOT NULL -- Nombre de la categoría
);
GO

-- 4. Crear la tabla "Hija" (Los libros)
CREATE TABLE PUBLICACION (
    idpublicacion INT PRIMARY KEY IDENTITY(1,1), -- Se numera solo (1, 2, 3...)
    titulo VARCHAR(100) NOT NULL,
    autor VARCHAR(100),
    precio MONEY,
    stock INT,
    idtipo CHAR(3), -- Aquí se conecta con la tabla TIPO
    CONSTRAINT FK_TipoPublicacion FOREIGN KEY (idtipo) REFERENCES TIPO(idtipo)
);
GO
